Write a program that reads the radius r of a circle and prints the area formatted with 2 digits after the decimal point
Expected input:2
Expected output:12.57
%2%12.57
%using System;
class CirclePerimeterArea
{
    static void Main(string[] args)// p*r^2
    {
        double r = int.Parse(Console.ReadLine());
        double area = Math.PI * r * r;
        Console.WriteLine("{0:F2}",area);
    }
}
